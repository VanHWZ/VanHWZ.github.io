Reverse Nodes in k-Group

Original page

My solution

Traverse the list. For every k nodes, compute reversed list and concatenate it back.

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseGroup(self, head, tail):
        current = head
        prev = tail.next
        while current != tail:
            temp = current.next
            current.next = prev
            prev = current
            current = temp
        current.next = prev
        return current, head
            
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        start = ListNode(-1, head)
        current = start
        group_head = current.next
        group_tail = start
        
        cnt = 0
        while current.next:
            cnt += 1
            current = current.next
            
            if cnt % k == 0:                    
                group_tail.next, current = self.reverseGroup(group_head, current)
                group_tail = group_head
                group_head = current.next
                cnt = 0
                
        return start.next

Analysis

Time: O(n)

Space: O(1)
